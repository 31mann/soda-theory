5    def input_matrix():
6        rows = int(input("Enter the number of rows: "))
7        cols = int(input("Enter the number of columns: "))
8        matrix = []

10           row = []
11           for j in range(cols):
12               value = int(input(f"Enter the value for ({i + 1}, {j + 1}): "))
